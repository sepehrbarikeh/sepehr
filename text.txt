"use client";

import { useEffect, useRef, useState } from "react";
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";

gsap.registerPlugin(ScrollTrigger);

const NAV_ITEMS = ["Home", "About", "Projects", "Skills", "Experience", "Contact"];

export default function Navbar() {
  const [activeItem, setActiveItem] = useState("Home");
  const navContainerRef = useRef<HTMLDivElement>(null);
  const navInnerRef = useRef<HTMLElement>(null);
  const indicatorRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    // حالت اولیه: ناوبار کاملاً پنهان و در بالای صفحه
    gsap.set(navContainerRef.current, { y: -100, opacity: 0 });

    const ctx = gsap.context(() => {
      
      // انیمیشن اول: ورود ناوبار به صورت تمام‌عرض دقیقاً در پایان لندینگ پیج (280vh)
      ScrollTrigger.create({
        trigger: document.body,
        // محاسبات دقیق: ۲.۸ برابر ارتفاع مانیتور منهای ۱۰۰ پیکسل برای ورود نرم‌تر
        start: () => `top -${(window.innerHeight * 2.8) - 100}px`, 
        onEnter: () => {
          gsap.to(navContainerRef.current, {
            y: 0,
            opacity: 1,
            duration: 0.4,
            ease: "power2.out",
          });
        },
        onLeaveBack: () => {
          // وقتی کاربر به داخل لندینگ پیج برمی‌گردد، ناوبار دوباره پنهان شود
          gsap.to(navContainerRef.current, {
            y: -100,
            opacity: 0,
            duration: 0.3,
            ease: "power2.in",
          });
        }
      });

      // انیمیشن دوم: تبدیل ناوبار تمام‌عرض به کپسول (کمی بعد از وارد شدن ناوبار)
      gsap.to(navInnerRef.current, {
        maxWidth: "1024px",              // جمع شدن از عرض
        marginTop: "16px",               // فاصله از سقف
        borderRadius: "9999px",          // کپسولی شدن
        backgroundColor: "rgba(10, 10, 12, 0.85)", // شیشه‌ای شدن
        backdropFilter: "blur(16px)",
        paddingLeft: "32px",
        paddingRight: "32px",
        boxShadow: "0 10px 30px rgba(0, 0, 0, 0.5)",
        borderColor: "rgba(255, 255, 255, 0.08)",
        duration: 0.35,
        scrollTrigger: {
          trigger: document.body,
          // شروع کپسول شدن: ۱۵۰ پیکسل بعد از اینکه کاملاً لندینگ ۲.۸ را رد کردیم
          start: () => `top -${(window.innerHeight * 2.8) + 150}px`, 
          toggleActions: "play none none reverse", // برگشت به بالا = تبدیل دوباره به حالت تخت و تمام‌عرض
        }
      });

    }, navContainerRef);

    return () => ctx.revert();
  }, []);

  // انیمیشن سُر خوردن خط بنفش زیر منوها
  const handleItemClick = (item: string, e: React.MouseEvent<HTMLButtonElement>) => {
    setActiveItem(item);
    const target = e.currentTarget;
    if (indicatorRef.current) {
      gsap.to(indicatorRef.current, {
        x: target.offsetLeft,
        width: target.offsetWidth,
        duration: 0.35,
        ease: "power2.out"
      });
    }
  };

  return (
    <div ref={navContainerRef} className="fixed top-0 left-0 right-0 z-50 w-full px-0 transition-all duration-300">
      <nav
        ref={navInnerRef}
        className="mx-auto flex w-full max-w-7xl items-center justify-between border-b border-white/5 bg-[#0a0a0c]/90 py-5 px-6 md:px-12 transition-all duration-300"
      >
        {/* لوگو */}
        <div className="flex items-center space-x-2 space-x-reverse text-lg font-bold text-white select-none">
          <span className="text-purple-500 font-mono">{"</>"}</span>
          <span className="tracking-tight">Sepehr</span>
        </div>

        {/* منوها */}
        <div className="relative hidden items-center space-x-1 space-x-reverse md:flex">
          {NAV_ITEMS.map((item) => (
            <button
              key={item}
              onClick={(e) => handleItemClick(item, e)}
              className={`px-4 py-2 text-sm font-medium transition-colors duration-300 ${
                activeItem === item ? "text-white" : "text-zinc-400 hover:text-zinc-200"
              }`}
            >
              {item}
            </button>
          ))}
          <div ref={indicatorRef} className="absolute bottom-0 h-[2px] bg-purple-500 rounded-full" style={{ left: "16px", width: "42px" }} />
        </div>

        {/* دکمه‌ها */}
        <div className="flex items-center space-x-4 space-x-reverse">
          
          <button className="rounded-xl bg-purple-600 px-5 py-2 text-sm font-semibold text-white shadow-[0_4px_20px_rgba(147,51,234,0.3)] transition-all duration-300 hover:scale-105 hover:bg-purple-500">
          Let's Talk
          </button>
        </div>
      </nav>
    </div>
  );
}

"use client";

import { useEffect, useRef } from "react";
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";
import heroImage from "@/public/hero-image.png";
import IntroText from "./introText";
import Image from "next/image";

gsap.registerPlugin(ScrollTrigger);

export default function HeroSection() {
    // ۱. فقط یک رفرنس برای کل سکشن می‌سازیم
    const mainContainerRef = useRef<HTMLDivElement>(null);

    useEffect(() => {
        // ۲. با پاس دادن mainContainerRef به عنوان آرگومان دوم gsap.context، مکانیزم Scope فعال می‌شود
        const ctx = gsap.context(() => {

            const tl = gsap.timeline({
                scrollTrigger: {
                    trigger: mainContainerRef.current,
                    start: "top top",
                    end: "bottom bottom",
                    scrub: 1.2,
                },
            });

            // ۳. حالا خیلی شیک و راحت از کلاس‌های معمولی CSS استفاده می‌کنیم!
            tl.fromTo(".hero-text", // به جای رفرنس، کلاس دادیم
                { x: -300, opacity: 0 },
                { x: 0, opacity: 1, ease: "power2.out" },
                0
            )
                .fromTo(".hero-media",
                    { x: 300, opacity: 0, scale: 0.9 },
                    { x: 0, opacity: 1, scale: 1, ease: "power2.out" },
                    0
                )
                .to(".scroll-down", {
                    opacity: 1,
                    y: 20,
                    duration: 0.2
                }, 0.1);

        }, mainContainerRef); // 👈 این همان جادوی Scope است

        return () => ctx.revert();
    }, []);

    return (
        <section
            id="hero-section"
            ref={mainContainerRef} // رفرنس مادر
            className="relative h-[280vh] w-full bg-[#00000c]"
        >
            <div className="sticky top-0 flex h-screen w-full flex-col items-center justify-center overflow-hidden px-6 lg:flex-row lg:px-16">

                <div className="hero-media mt-12 flex w-full items-center justify-center lg:mt-0 lg:w-1/2">
                    <Image src={heroImage} alt="Hero Image" />
                </div>
                {/* متن‌ها (فقط کلاس hero-text را به آن دادیم) */}
                <div className="hero-text max-w-2xl text-center text-white lg:text-left">
                    <IntroText />
                </div>
                {/* دکمه اسکرول (کلاس scroll-down) */}
                <div className="scroll-down absolute bottom-8 left-6 flex items-center space-x-2 space-x-reverse text-xs text-zinc-500 font-mono select-none md:left-16">
                    <span>Scroll Down</span>
                    <span className="animate-bounce">↓</span>
                </div>

            </div>
        </section>
    );
}
